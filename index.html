// ========================================================================== // [FWU] Mayesh Weekly Order System - 雙標籤獨立分流讀取專用後端 GAS // ========================================================================== function doGet(e) { var action = e.parameter.action; if (action === 'init') { var ss = SpreadsheetApp.getActiveSpreadsheet(); // 1. 【核心鎖定】 專門讀取 "Price List" 分頁（純文字 A 欄至 F 欄結構） var priceListSheet = ss.getSheetByName("Price List"); var priceData = priceListSheet ? priceListSheet.getDataRange().getValues() : []; var priceListArray = []; for (var i = 1; i < priceData.length; i++) { var row = priceData[i]; if (row[0] && row[0].toString().trim() !== "") { priceListArray.push([ row[0] ? row[0].toString().trim() : "", // [0] Variety Name (A欄) row[1] ? row[1].toString().trim() : "", // [1] Scientific Name (B欄) row[2] ? row[2].toString().trim() : "", // [2] Color (C欄) row[3] ? row[3].toString().trim() : "", // [3] Stem Length (D欄) row[4] ? row[4].toString().trim() : "", // [4] FOB_USD (E欄) row[5] ? row[5].toString().trim() : "" // [5] Memo (F欄) ]); } } // 2. 【核心鎖定】 專門讀取 "Variety List" (或 Varieties) 分頁（包含圖片與分類結構） var varietySheet = ss.getSheetByName("Variety List") || ss.getSheetByName("Varieties"); var varietyData = varietySheet ? varietySheet.getDataRange().getValues() : []; var varietiesArray = []; for (var j = 1; j < varietyData.length; j++) { var vRow = varietyData[j]; if (vRow[0] && vRow[0].toString().trim() !== "") { varietiesArray.push([ vRow[0] ? vRow[0].toString().trim() : "", // [0] Variety Name vRow[1] ? vRow[1].toString().trim() : "", // [1] Intro / Description vRow[2] ? vRow[2].toString().trim() : "", // [2] Image URL (含下單大圖) vRow[3] ? vRow[3].toString().trim() : "" // [3] Category (大分類，如 Oncidium 等) ]); } } // 3. 讀取 Pack Rate 箱裝規格參考表 var packRateSheet = ss.getSheetByName("Pack Rate"); var packRateData = packRateSheet ? packRateSheet.getDataRange().getValues() : []; // 封裝並回傳給前端網頁 var result = { priceList: priceListArray, // 鎖定給 Price list 分頁 varieties: varietiesArray, // 鎖定給 Place order 分頁（含大圖） packRate: packRateData }; return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } if (action === 'getOrders') { var ss = SpreadsheetApp.getActiveSpreadsheet(); var orderSheet = ss.getSheetByName("Orders"); var orderData = orderSheet ? orderSheet.getDataRange().getValues() : []; return ContentService.createTextOutput(JSON.stringify(orderData)) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var orderSheet = ss.getSheetByName("Orders"); if (!orderSheet) { orderSheet = ss.insertSheet("Orders"); orderSheet.appendRow(["Timestamp", "Location", "Shipping Date", "Variety Name", "Qty / Unit"]); } var postData = JSON.parse(e.postData.contents); var timestamp = new Date(); var location = postData.location; var shippingDate = postData.shippingDate; var items = postData.items; for (var i = 0; i < items.length; i++) { var item = items[i]; orderSheet.appendRow([timestamp, location, shippingDate, item.name, item.qty + " " + item.unit]); } return ContentService.createTextOutput(JSON.stringify({status: "success"})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status: "error", message: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
